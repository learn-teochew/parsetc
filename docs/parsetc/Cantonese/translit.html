126    def _lookup_terminal(self, items, which="initial"):
127        """Define method to look up terminals from dictionary
128
129        This is an internal method used by the rule-specific methods.

131        Arguments
132        ---------
133        items : list
134            Children from parse tree

136            Key for terminals dictionary to look up

138        Returns

140        dict
141            Dictionary of terminal strings keyed by type

143        trdict = {
144            term: TERMINALS[which][term][self.system]
145            for term in TERMINALS[which]
146            if self.system in TERMINALS[which][term]

150                merged_to = MERGERS[which][term][self.system]
151                trdict[term] = TERMINALS[which][merged_to][self.system]

198    def _lookup_terminal(self, items, which="initial"):
199        """Define method to look up terminals from dictionary
200
201        This is an internal method used by the rule-specific methods. In
202        addition to the child elements it returns name of the rule itself, as
203        the elements have to be rearranged in the parent.
204
205        Arguments
206        ---------
207        items : list
208            Children from parse tree
209        which : str
210            Key for terminals dictionary to look up
211
212        Returns
213        -------
214        Tuple of:
215
216        str
217            The rule key name
218        dict
219            Dictionary of terminal strings keyed by type
220        """
221        trdict = {
222            term: TERMINALS[which][term][self.system]
223            for term in TERMINALS[which]
224            if self.system in TERMINALS[which][term]
225        }
226        for term in MERGERS[which]:
227            if self.system in MERGERS[which][term]:
228                merged_to = MERGERS[which][term][self.system]
229                trdict[term] = TERMINALS[which][merged_to][self.system]
230        return which, trdict[items[0].type]

258    def syllable_tone(self, items):
259        """Syllable with tone diacritics
260
261        Yale romanization uses diacritics instead of tone numbers, as well as
262        tone spelling (-h to medial for tones 4,5,6,9). For tone 9 the -h comes
263        before the stop coda, therefore the elements have to be rearranged.
264
265        Returns
266        -------
267        str
268            Syllable in Yale romanization with tone diacritics, Unicode NFC
269            normalized.
270        """
271        trdict = {
272            "tone_1a": "\u0304",
273            "tone_1b": "\u0300",
274            "tone_2": "\u0301",
275            "tone_3": "",
276            "tone_4": "\u0300",
277            "tone_5": "\u0301",
278            "tone_6": "",
279            "tone_7": "\u0304",
280            "tone_8": "",
281            "tone_9": "",
282        }
283        # items is a list of: [initial] final tone_citation
284        initial = items[0]  # first element, either tuple or None
285        tone = items[-1][0]  # last element
286        final_text = ""
287        if items[1][0] == "final":
288            final = items[1][1]
289            if "medial" in final:
290                final_text += final["medial"] + trdict[tone]
291                if tone in ["tone_4", "tone_5", "tone_6"]:
292                    final_text += "h"
293                if "codanasal" in final:
294                    final_text += final["codanasal"]
295            else:  # codanasalnasal only
296                final_text += final["codanasal"] + trdict[tone]
297        elif items[1][0] == "final_entering":
298            final = items[1][1]
299            final_text += final["medial"] + trdict[tone]
300            if tone == "tone_9":
301                final_text += "h"
302            final_text += final["codastop"]
303        syllable_text = ""
304        if initial is not None:
305            syllable_text += initial[1]
306        syllable_text += final_text
307        return unicodedata.normalize("NFC", syllable_text)

200        """Define method to look up terminals from dictionary
201
202        This is an internal method used by the rule-specific methods. In
203        addition to the child elements it returns name of the rule itself, as

216

219        dict
220            Dictionary of terminal strings keyed by type

275            "tone_2": "\u0301",
276            "tone_3": "",

279            "tone_6": "",
280            "tone_7": "\u0304",
281            "tone_8": "",
282            "tone_9": "",
283        }
284        # items is a list of: [initial] final tone_citation
285        initial = items[0]  # first element, either tuple or None
286        tone = items[-1][0]  # last element
287        final_text = ""
288        if items[1][0] == "final":
289            final = items[1][1]
290            if "medial" in final:

292                if tone in ["tone_4", "tone_5", "tone_6"]:

299            final = items[1][1]
300            final_text += final["medial"] + trdict[tone]
301            if tone == "tone_9":
302                final_text += "h"
303            final_text += final["codastop"]
304        syllable_text = ""
305        if initial is not None:
306            syllable_text += initial[1]
307        syllable_text += final_text
308        return unicodedata.normalize("NFC", syllable_text)
