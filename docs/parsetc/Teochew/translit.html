119
120        This is an internal method used by the rule-specific methods.

246    def syllable_tone(self, items):
247        """Syllable with tone diacritics
248
249        Duffus romanization uses diacritics instead of tone numbers. Diacritics
250        are placed on vowels in the medial; otherwise on the nasal -m- or -n-.
251
252        Returns
253        -------
254        str
255            Syllable in Duffus romanization with tone diacritics, Unicode NFC
256            normalized.
257        """
258        trdict = {
259            "1": "",
260            "2": "\u0301",
261            "3": "\u0300",
262            "4": "",
263            "5": "\u0302",
264            "6": "\u0303",
265            "7": "\u0304",
266            "8": "\u0307",
267            "0": "",
268        }
269        syllab = "".join([str_or_None(i) for i in items[:-1]])  # syllable without tone
270        tone = items[-1]
271        firstvowel = re.search(r"[aeiou]", syllab)
272        if firstvowel:
273            # put tone mark on first vowel letter
274            inspos = firstvowel.span()[1]
275        else:
276            # no vowel in syllable, put on nasal codas n or m, n comes first
277            firstnasal = re.search(r"[nm]", syllab)
278            inspos = firstnasal.span()[1]
279        syllab = syllab[0:inspos] + trdict[tone] + syllab[inspos:]
280        syllab = unicodedata.normalize("NFC", syllab)
281        return syllab
282
283    def word_sep(self, items):
284        # replace all syllable separators with hyphens
285        # and separate syllables with hyphens if no
286        # syllable separator is present
287        return "-".join([i for i in items if i != "-"])
